- cd "navegar entre
- cd .. "voltar entre
- git init "criar init
- git commit -m + mais nota de atualização "clean commit"

- git branch "ver branch's
- git config --list "listar dados de configuração
- git config --global --unset user.emal "trocar ou atualizar email
